
\qpreamble{
    Consider this block of code:
    
\code{
1  x = roll() \\
2  y = roll() \\
3  if x >= 4: \\
4  \mytab   x = x - 2 \\
5  if y < 4: \\
6  \mytab   y = y + 4 \\
7  Tile(PINK).place(x,y)
}
    % Create a list of rolls to choose from. Each list element represents a new quiz version.
    \def\combos{{1,3},{3,2},{2,1},{4,6},{5,4},{6,5},{1,2},{3,1},{2,3},{4,6}}

}

\qversion{a}{
    % create a list of x,y rolls for different versions so that exactly 1 if is true
    % true for x is 4,5,6   true for y is 1,2,3

    % get the next list in combos, then isolate the first and second values
    % getitem is a function in main.tex. Argument #1 is the list. #2 is the index to retreive. #3 variable it will be stored in
    \getitem{\combos}{\the\value{versioncount}}{combo}
    \getitem{\combo}{1}{xval}
    \getitem{\combo}{2}{yval}

    % debugging: THE INDEX \arabic{versioncount} with combo \combo

    \begin{quote}   % indent
    Use \underline{   \xval{}   } as the value of the roll at line 1.  \\
    Use \underline{   \yval{}   } as the value of the roll at line 2.  \\
    \end{quote}
    
    What is the value of x at line 7 ? \rule{35mm}{0.15mm}
    
    \vspace{8mm}
    
    What is the value of y at line 7 ? \rule{35mm}{0.15mm}
}
